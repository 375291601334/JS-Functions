JS Functions

A JavaScript function is a block of code designed to perform a particular task.

Function Declaration

function showGreeting(name, age) {
  alert("Hello! I’m"  + name + ", I’m " + age + ". Nice to meet you!");
};

showGreeting("Kitty", 5); // Hello! I'm Kitty, I'm 5. Nice to meet you!

Function starting with…

  • "get…" – return a value,
  • "calc…" – calculate something,
  • "create…" – create something,
  • "check…" – check something and return a boolean,
  • ...

Examples of such names:

  • showMessage(..) - shows a message
  • getAge(..) - returns the age (gets it somehow)
  • calcSum(..) - calculates and returns a sum
  • createForm(..) - creates a form
  • checkPermission(..) - checks a permission, returns true/false

Function Expression

let showGreeting = function (name, age) {
  alert("Hello! I’m " + name + ", I’m " + age + ". Nice to meet you!");
}

showGreeting("Porgi",98);

Named Function Expression

let showGreetingMessage = function showGreeting(name, age) {
  alert("Hello! I’m " + name + ", I’m " + age + ". Nice to meet you!");
};

showGreetingMessage("Mrs Hamster", 25); // Hello! I'm Mrs Hamster, I'm 25. Nice to meet you!

NFE

let showGreetingMessage = function internalGreeting(name, age) {
  if (name) {
    alert("Hello! I’m " + name + ", I’m " + age + ". Nice to meet you!");
  } else  {
    internalGreeting("anonymous", 0);
  }
};
let sayHi = showGreetingMessage;
showGreetingMessage = null;
sayHi();

local variables

function showMessage() {
  let food = "apples"; // local variable
  alert( "I like " + food + "!"" );
};
showMessage(); // I like apples
alert( food ); // Error! The variable is local to the function

Outer variables

let food = 'fish'; //outer variable
 
function showMessage() {
  alert(‘I like ’ + food +’!’);
}
 
showMessage(); // I like fish!

Shadowing

let food = "apples";
 
function showMessage() {
 let food = "fish"; // declare a local variable
 alert(‘I like ’ + food +’!’);
}
 
// the function will create and use its own variable food
showMessage(); // I like fish!
 
alert( food ); // apples

Shadowing

function showGreeting(name, age) {
  name = name + "the Great";
  alert("Hello! I’m"  + name + ", I’m " + age + ". Nice to meet you!");
};
let name = "Porgi";
showGreeting(name, 98);
 
alert( name ); // Porgi

Default values

function showGreeting(name, age) {
  alert("Hello! I’m "  + name + ", I’m " + age + ". Nice to meet you!");
};

showGreeting("Kitty"); 
// Hello! I’m Kitty, I’m UNDEFINED. Nice to meet you!

Default values

function showGreeting(name, age = 18) {
  alert("Hello! I’m "  + name + ", I’m " + age + ". Nice to meet you!");
};

showGreeting("Kitty"); // Hello! I’m Kitty, I’m 18. Nice to meet you!

Returning a value

function sum(a, b) {
  return a + b;
};
let result = sum(1, 2);
alert( result ); // 3

let c = 0;
function sum(a, b) {
 c = a + b;
